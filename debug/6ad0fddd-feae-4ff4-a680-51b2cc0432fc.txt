// min returns the minimum of three integers
func min(a, b, c int) int {
	if a < b {
		if a < c {
			return a
		}
		return c
	}
	if b < c {
		return b
	}
	return c
}

// levenshteinDistance calculates the edit distance between two strings
// Used for fuzzy matching command names to suggest corrections for typos
func levenshteinDistance(s1, s2 string) int {
	if len(s1) == 0 {
		return len(s2)
	}
	if len(s2) == 0 {
		return len(s1)
	}

	matrix := make([][]int, len(s1)+1)
	for i := range matrix {
		matrix[i] = make([]int, len(s2)+1)
		matrix[i][0] = i
	}
	for j := range matrix[0] {
		matrix[0][j] = j
	}

	for i := 1; i <= len(s1); i++ {
		for j := 1; j <= len(s2); j++ {
			cost := 0
			if s1[i-1] != s2[j-1] {
				cost = 1
			}
			matrix[i][j] = min(
				matrix[i-1][j]+1,      // deletion
				matrix[i][j-1]+1,      // insertion
				matrix[i-1][j-1]+cost, // substitution
			)
		}
	}

	return matrix[len(s1)][len(s2)]
}

// suggestCommand suggests a similar command when user makes a typo
// Returns a helpful suggestion string or fallback help message
func (h *SlackEventsHandler) suggestCommand(attempted string, isAdmin bool) string {
	commands := []string{"log", "list", "edit", "admin", "help"}
	adminCommands := []string{
		"add-client", "list-clients", "remove-client",
		"list-users", "make-admin", "assign", "unassign",
	}

	attempted = strings.ToLower(strings.TrimSpace(attempted))

	// Check main commands
	var bestMatch string
	var bestDistance int = 999

	for _, cmd := range commands {
		distance := levenshteinDistance(cmd, attempted)
		if distance < bestDistance && distance <= 2 {
			bestDistance = distance
			bestMatch = cmd
		}
	}

	if bestMatch != "" {
		return fmt.Sprintf("\n\n:bulb: Did you mean `/cromie %s`?", bestMatch)
	}

	// Check admin commands if user is admin
	if isAdmin {
		bestDistance = 999
		for _, cmd := range adminCommands {
			distance := levenshteinDistance(cmd, attempted)
			if distance < bestDistance && distance <= 2 {
				bestDistance = distance
				bestMatch = cmd
			}
		}

		if bestMatch != "" {
			return fmt.Sprintf("\n\n:bulb: Did you mean `/cromie admin %s`?", bestMatch)
		}
	}

	return "\n\nType `/cromie help` to see all available commands."
}
